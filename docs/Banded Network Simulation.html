# Simulate multivariate normal and check empirical covariances

def _new_sample(n_samples, n_features, cov):
    X = prng.multivariate_normal(np.zeros(n_features), cov, size=n_samples)
    return X

X1 = _new_sample(1000,n_features,covariance)
X2 = _new_sample(50,n_features,covariance)
